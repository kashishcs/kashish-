// Example: bubble sort step generator (simplified)
function* bubbleSteps(arr) {
  let a = arr.slice();
  for (let i = 0; i < a.length; i++) {
    for (let j = 0; j < a.length - 1 - i; j++) {
      if (a[j] > a[j+1]) {
        [a[j], a[j+1]] = [a[j+1], a[j]];
        yield a.slice();
      }
    }
  }
  return a;
}

A tool that compares documents using Jaccard similarity and sequence matching to highlight overlaps. The backend prototype uses Python and difflib; the front-end demo below shows a simple token overlap check. 

// Simple Jaccard similarity (JS)
function jaccard(a, b) {
  const A = new Set(a.split(/\s+/));
  const B = new Set(b.split(/\s+/));
  const inter = new Set([...A].filter(x => B.has(x)));
  const union = new Set([...A, ...B]);
  return inter.size / union.size;
}
